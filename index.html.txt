<script>

/* =====================
   面板控制
===================== */
const panel = document.getElementById('panel');

document.getElementById('gear').onclick = () => {
  panel.classList.add('show');
};

function save(){
  localStorage.setItem('salary', salary.value);
  localStorage.setItem('factor', factor.value);
  localStorage.setItem('start', start.value);
  localStorage.setItem('end', end.value);

  panel.classList.remove('show');
}

/* =====================
   计时核心
===================== */
function tick(){
  const now = new Date();
  document.getElementById('now').innerText =
    now.toLocaleTimeString();

  const salary = +localStorage.getItem('salary') || 0;
  const factor = +localStorage.getItem('factor') || 1;
  const start = localStorage.getItem('start') || "09:00";
  const end = localStorage.getItem('end') || "18:00";

  const [sh,sm] = start.split(':').map(Number);
  const [eh,em] = end.split(':').map(Number);

  const s = new Date();
  s.setHours(sh,sm,0);

  const e = new Date();
  e.setHours(eh,em,0);

  const real = salary * factor;
  const workSec = (e - s) / 1000;
  const perSec = real / (22 * workSec);

  let money = 0;
  let status = "";

  if(now < s){
    money = 0;
    status = "未上班";
  }else if(now > e){
    money = perSec * workSec;
    status = "已下班";
  }else{
    money = perSec * ((now - s)/1000);
    status = "工作中";
  }

  document.getElementById('money').innerText =
    money.toFixed(2);

  document.getElementById('status').innerText = status;
}

setInterval(tick,1000);

/* =====================
   初始化
===================== */
window.onload = () => {
  ['salary','factor','start','end'].forEach(k=>{
    const el = document.getElementById(k);
    if(localStorage.getItem(k)) el.value = localStorage.getItem(k);
  });
};
